18  package com.orangesignal.csv.filters;
19  
20  import java.util.List;
21  
22  /**
23   * 指定された項目名に対応する区切り文字形式データの値が下限値から上限値の範囲かどうかでフィルタを適用する区切り文字形式データフィルタの実装です。
24   * 
25   * @author 杉澤 浩二
26   * @since 1.2.3
27   */
28  public class ColumnNameBetweenExpression extends ColumnNameExpression {
29  
30  	/**
31  	 * 下限値を保持します。
32  	 */
33  	private String low;
34  
35  	/**
36  	 * 上限値を保持します。
37  	 */
38  	private String high;
39  
40  	/**
41  	 * コンストラクタです。
42  	 * 
43  	 * @param name 項目名
44  	 * @param low 下限値
45  	 * @param high 上限値
46  	 * @throws IllegalArgumentException パラメータが <code>null</code> の場合
47  	 */
48  	protected ColumnNameBetweenExpression(final String name, final String low, final String high) {
49  		super(name);
50  		if (low == null || high == null) {
51  			throw new IllegalArgumentException("Low or High must not be null");
52  		}
53  		this.low = low;
54  		this.high = high;
55  	}
56  
57  	@Override
58  	public boolean accept(final List<String> header, final List<String> values) {
59  		final int position = header.indexOf(name);
60  		if (position == -1) {
61  			throw new IllegalArgumentException(String.format("Invalid column name %s", name));
62  		}
63  		return CsvExpressionUtils.between(values, position, low, high);
64  	}
65  
66  }
