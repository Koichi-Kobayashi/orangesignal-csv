18  package com.orangesignal.csv.filters;
19  
20  import java.io.IOException;
21  import java.util.Comparator;
22  
23  /**
24   * 指定された Java プログラム要素のフィールド値が下限値から上限値の範囲かどうかでフィルタを適用する Java プログラム要素フィルタの実装です。
25   * 
26   * @author 杉澤 浩二
27   * @since 1.2.3
28   */
29  public class BeanBetweenExpression extends BeanExpression {
30  
31  	/**
32  	 * 下限値を保持します。
33  	 */
34  	private Object low;
35  
36  	/**
37  	 * 上限値を保持します。
38  	 */
39  	private Object high;
40  
41  	/**
42  	 * コンパレータを保持します。
43  	 */
44  	@SuppressWarnings("rawtypes")
45  	private Comparator comparator;
46  
47  	/**
48  	 * コンストラクタです。
49  	 * 
50  	 * @param name フィールド名
51  	 * @param low 下限値
52  	 * @param high 上限値
53  	 * @throws IllegalArgumentException パラメータが <code>null</code> の場合
54  	 */
55  	protected BeanBetweenExpression(final String name, final Object low, final Object high) {
56  		this(name, low, high, null);
57  	}
58  
59  	/**
60  	 * コンストラクタです。
61  	 * 
62  	 * @param name フィールド名
63  	 * @param low 下限値
64  	 * @param high 上限値
65  	 * @param comparator コンパレータ (オプション)
66  	 * @throws IllegalArgumentException パラメータが <code>null</code> の場合
67  	 */
68  	protected BeanBetweenExpression(final String name, final Object low, final Object high, @SuppressWarnings("rawtypes") final Comparator comparator) {
69  		super(name);
70  		if (low == null || high == null) {
71  			throw new IllegalArgumentException("Low or High must not be null");
72  		}
73  		this.low = low;
74  		this.high = high;
75  		this.comparator = comparator;
76  	}
77  
78  	@Override
79  	public boolean accept(final Object bean) throws IOException {
80  		return BeanExpressionUtils.between(bean, name, low, high, comparator);
81  	}
82  
83  }
