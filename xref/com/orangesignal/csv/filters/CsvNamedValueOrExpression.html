18  package com.orangesignal.csv.filters;
19  
20  import java.util.List;
21  
22  /**
23   * 指定された区切り文字形式データの項目名リストと値リストでフィルタする区切り文字形式データフィルタ群の論理和でフィルタを適用する区切り文字形式データフィルタの実装です。
24   * 
25   * @author 杉澤 浩二
26   * @since 1.2.3
27   */
28  public class CsvNamedValueOrExpression extends CsvNamedValueLogicalExpression {
29  
30  	/**
31  	 * デフォルトコンストラクタです。
32  	 */
33  	public CsvNamedValueOrExpression() {
34  		super();
35  	}
36  
37  	/**
38  	 * コンストラクタです。
39  	 * 
40  	 * @param filters 区切り文字形式データフィルタ群
41  	 * @throws IllegalArgumentException <code>filters</code> が <code>null</code> の場合
42  	 */
43  	protected CsvNamedValueOrExpression(final CsvNamedValueFilter... filters) {
44  		super(filters);
45  	}
46  
47  	@Override
48  	public boolean accept(final List<String> header, final List<String> values) {
49  		if (filters.isEmpty()) {
50  			throw new IllegalArgumentException("Filters must not be empty");
51  		}
52  		for (final CsvNamedValueFilter filter : filters) {
53  			if (filter.accept(header, values)) {
54  				return true;
55  			}
56  		}
57  		return false;
58  	}
59  
60  }
