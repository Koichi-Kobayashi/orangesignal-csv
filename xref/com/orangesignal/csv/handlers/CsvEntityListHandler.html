18  package com.orangesignal.csv.handlers;
19  
20  import java.io.IOException;
21  import java.util.ArrayList;
22  import java.util.List;
23  
24  import com.orangesignal.csv.CsvReader;
25  import com.orangesignal.csv.CsvWriter;
26  import com.orangesignal.csv.annotation.CsvEntity;
27  import com.orangesignal.csv.bean.CsvEntityOperation;
28  import com.orangesignal.csv.bean.CsvEntityTemplate;
29  import com.orangesignal.csv.filters.CsvNamedValueFilter;
30  import com.orangesignal.csv.io.CsvEntityReader;
31  import com.orangesignal.csv.io.CsvEntityWriter;
32  
33  /**
34   * 区切り文字形式データ注釈要素 {@link com.orangesignal.csv.annotation.CsvEntity} で注釈付けされた Java プログラム要素のリストで区切り文字形式データアクセスを行うハンドラを提供します。
35   * 
36   * @author 杉澤 浩二
37   * @see com.orangesignal.csv.annotation.CsvEntity
38   * @see com.orangesignal.csv.annotation.CsvColumn
39   * @see com.orangesignal.csv.annotation.CsvColumns
40   * @see com.orangesignal.csv.manager.CsvEntityManager
41   */
42  public class CsvEntityListHandler<T> extends AbstractBeanListHandler<T, CsvEntityTemplate<T>, CsvEntityListHandler<T>> implements CsvEntityOperation<CsvEntityListHandler<T>> {
43  
44  	/**
45  	 * コンストラクタです。
46  	 * 
47  	 * @param entityClass 区切り文字形式データ注釈要素 {@link com.orangesignal.csv.annotation.CsvEntity} で注釈付けされた Java プログラム要素の型
48  	 * @throws IllegalArgumentException {@code entityClass} が {@code null} または不正な場合
49  	 */
50  	public CsvEntityListHandler(final Class<T> entityClass) {
51  		super(CsvEntityTemplate.newInstance(entityClass));
52  	}
53  
54  	@Override
55  	public CsvEntityListHandler<T> filter(final CsvNamedValueFilter filter) {
56  		template.filter(filter);
57  		return this;
58  	}
59  
60  	@Override
61  	public List<T> load(final CsvReader reader, final boolean ignoreScalar) throws IOException {
62  		@SuppressWarnings("resource")
63  		final CsvEntityReader<T> r = new CsvEntityReader<T>(reader, template);
64  
65  		// すべてのデータを読取って繰返し処理します。
66  		final List<T> results = new ArrayList<T>();
67  		final boolean order = ignoreScalar || (orders != null && !orders.isEmpty());
68  		int offset = 0;
69  
70  		List<String> values;
71  		while ((values = r.readValues()) != null && (order || limit <= 0 || results.size() < limit)) {
72  			if (beanFilter == null && !order && offset < this.offset) {
73  				offset++;
74  				continue;
75  			}
76  			final T entity = r.toEntity(values);
77  			if (beanFilter != null) {
78  				if (!beanFilter.accept(entity)) {
79  					continue;
80  				}
81  				if (!order && offset < this.offset) {
82  					offset++;
83  					continue;
84  				}
85  			}
86  			results.add(entity);
87  		}
88  
89  		if (ignoreScalar || !order) {
90  			return results;
91  		}
92  		return processScalar(results);
93  	}
94  
95  	@Override
96  	public void save(final List<T> entities, final CsvWriter writer) throws IOException {
97  		if (entities == null) {
98  			throw new IllegalArgumentException("CsvEntities must not be null");
99  		}
100 
101 		@SuppressWarnings("resource")
102 		final CsvEntityWriter<T> w = new CsvEntityWriter<T>(writer, template);
103 
104 		// データ出力
105 		for (final T entity : entities) {
106 			if (entity == null || entity.getClass().getAnnotation(CsvEntity.class) == null) {
107 				w.write(null);
108 				continue;
109 			} else if (beanFilter != null && !beanFilter.accept(entity)) {
110 				continue;
111 			}
112 
113 			w.write(entity);
114 		}
115 	}
116 
117 }
