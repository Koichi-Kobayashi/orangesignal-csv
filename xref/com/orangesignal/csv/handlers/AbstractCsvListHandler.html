18  package com.orangesignal.csv.handlers;
19  
20  import java.io.IOException;
21  import java.util.List;
22  
23  import com.orangesignal.csv.CsvListHandler;
24  import com.orangesignal.csv.CsvReader;
25  
26  /**
27   * 区切り文字形式データリストのデータアクセスを行うハンドラの基底クラスを提供します。
28   *
29   * @param <T> 区切り文字形式データの型
30   * @param <H> 区切り文字形式データリストのデータアクセスハンドラの型
31   * @author 杉澤 浩二
32   * @since 1.3.0
33   */
34  public abstract class AbstractCsvListHandler<T, H extends AbstractCsvListHandler<T, H>> implements CsvListHandler<T> {
35  
36  	/**
37  	 * 取得データの開始位置を保持します。
38  	 */
39  	protected int offset;
40  
41  	/**
42  	 * 取得データの限度数を保持します。
43  	 */
44  	protected int limit;
45  
46  	/**
47  	 * デフォルトコンストラクタです。
48  	 */
49  	public AbstractCsvListHandler() {}
50  
51  	@Override
52  	public void setOffset(final int offset) {
53  		this.offset = offset;
54  	}
55  
56  	@SuppressWarnings("unchecked")
57  	@Override
58  	public H offset(final int offset) {
59  		setOffset(offset);
60  		return (H) this;
61  	}
62  
63  	@Override
64  	public void setLimit(final int limit) {
65  		this.limit = limit;
66  	}
67  
68  	@SuppressWarnings("unchecked")
69  	@Override
70  	public H limit(final int limit) {
71  		setLimit(limit);
72  		return (H) this;
73  	}
74  
75  	@Override
76  	public List<T> load(final CsvReader reader) throws IOException {
77  		return load(reader, false);
78  	}
79  
80  	/**
81  	 * この実装は単に {@code offset} と {@code limit} を使用して処理します。
82  	 */
83  	@Override
84  	public List<T> processScalar(final List<T> list) {
85  		final int fromIndex = Math.max(this.offset, 0);
86  		final int toIndex = this.limit <= 0 ? list.size() : Math.min(fromIndex + this.limit, list.size());
87  		return list.subList(fromIndex, toIndex);
88  	}
89  
90  }
