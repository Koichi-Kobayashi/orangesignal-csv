18  package com.orangesignal.csv.entryfilters;
19  
20  import java.io.Serializable;
21  
22  import com.orangesignal.csv.LhaEntryFilter;
23  import com.orangesignal.csv.ZipEntryFilter;
24  
25  /**
26   * エントリフィルタの基底クラスを提供します。
27   * 
28   * @author 杉澤 浩二
29   * @since 1.2.2
30   */
31  public abstract class AbstractEntryFilter implements ZipEntryFilter, LhaEntryFilter, Serializable {
32  
33  	private static final long serialVersionUID = -1492900359515513779L;
34  
35  	/**
36  	 * デフォルトコンストラクタです。
37  	 */
38  	protected AbstractEntryFilter() {}
39  
40  	@Override
41  	public String toString() {
42  		final String name = getClass().getName();
43  		final int period = name.lastIndexOf('.');
44  		return (period > 0 ? name.substring(period + 1) : name);
45  	}
46  
47  }
