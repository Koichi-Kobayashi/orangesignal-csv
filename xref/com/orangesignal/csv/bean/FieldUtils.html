17  package com.orangesignal.csv.bean;
18  
19  import java.io.IOException;
20  import java.lang.reflect.Field;
21  
22  /**
23   * Java プログラム要素のフィールド操作に関するユーティリティを提供します。
24   * 
25   * @author Koji Sugisawa
26   * @since 1.4.0
27   */
28  public abstract class FieldUtils {
29  
30  	/**
31  	 * インスタンス化できない事を強制します。
32  	 */
33  	protected FieldUtils() {}
34  
35  	// ------------------------------------------------------------------------
36  	// フィールド操作用静的メソッド群
37  
38  	/**
39  	 * 指定された Java プログラム要素の型が表すクラスの指定された宣言フィールドをリフレクトする {@link Field} オブジェクトを返します。
40  	 * 
41  	 * @param type Java プログラム要素の型
42  	 * @param name フィールド名
43  	 * @return 指定された Java プログラム要素の {@link Field} オブジェクト
44  	 * @throws IOException 指定された名前のフィールドが見つからない場合
45  	 * @throws NullPointerException {@code name} が {@code null} の場合
46  	 * @throws SecurityException 
47  	 */
48  	public static final Field getField(final Class<?> type, final String name) throws IOException {
49  		try {
50  			return type.getDeclaredField(name);
51  		} catch (final NoSuchFieldException e) {
52  			throw new IOException("Field " + name + " not found in " + type.getName() + ": " + e.getMessage(), e);
53  		}
54  	}
55  
56  	/**
57  	 * 指定された Java プログラム要素の指定されたフィールドを、指定された新しい値に設定します。
58  	 * 基本となるフィールドにプリミティブ型が指定されている場合、新しい値は自動的にラップ解除されます。
59  	 * 
60  	 * @param bean フィールドを変更する Java プログラム要素
61  	 * @param field フィールド
62  	 * @param value 変更中の Java プログラム要素の新しいフィールド値
63  	 * @throws IOException 基本となるフィールドにアクセスできない場合。または指定されたオブジェクトが基本となるフィールド (またはそのサブクラスか実装側) を宣言するクラスまたはインタフェースのインスタンスではない場合、あるいはラップ解除変換が失敗した場合
64  	 * @throws NullPointerException 指定されたオブジェクトが {@code null} で、フィールドがインスタンスフィールドの場合
65  	 * @throws SecurityException 
66  	 */
67  	public static final void setFieldValue(final Object bean, final Field field, final Object value) throws IOException {
68  		if (!field.isAccessible()) {
69  			field.setAccessible(true);
70  		}
71  		try {
72  			field.set(bean, value);
73  		} catch (final IllegalAccessException e) {
74  			throw new IOException("Cannot set " + field.getName() + ": " + e.getMessage(), e);
75  		} catch (final IllegalArgumentException e) {
76  			throw new IOException("Cannot set " + field.getName() + ": " + e.getMessage(), e);
77  		}
78  	}
79  
80  	/**
81  	 * 指定された Java プログラム要素について、指定された {@link Field} によって表されるフィールドの値を返します。
82  	 * プリミティブ型の場合、オブジェクト内に自動的に格納されてから返されます。
83  	 * 
84  	 * @param bean Java プログラム要素
85  	 * @param field フィールド
86  	 * @return Java プログラム要素 {@code bean} 内で表現される値。プリミティブ値は適切なオブジェクト内にラップされてから返される
87  	 * @throws IOException 基本となるフィールドにアクセスできない場合。指定されたオブジェクトが基本となるフィールド (またはそのサブクラスか実装側) を宣言するクラスまたはインタフェースのインスタンスではない場合
88  	 * @throws NullPointerException 指定されたオブジェクトが {@code null} で、フィールドがインスタンスフィールドの場合
89  	 * @throws SecurityException 
90  	 */
91  	public static final Object getFieldValue(final Object bean, final Field field) throws IOException {
92  		if (!field.isAccessible()) {
93  			field.setAccessible(true);
94  		}
95  		try {
96  			return field.get(bean);
97  		} catch (final IllegalAccessException e) {
98  			throw new IOException("Cannot get " + field.getName() + ": " + e.getMessage(), e);
99  		} catch (final IllegalArgumentException e) {
100 			throw new IOException("Cannot get " + field.getName() + ": " + e.getMessage(), e);
101 		}
102 	}
103 
104 }
